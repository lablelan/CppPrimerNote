    练习 2.11

    指出下面的语句是声明还是定义

    (a) extern int ix = 1024;
(b) int iy;
(c) extern int iz;

    #include <iostream>

extern int ix = 1024;       // 定义 编译器会警告
extern int iz;              // 声明

int main()
{
    //extern int ix = 1024;     // 函数内会报错
    std::cout << ix << std::endl;
    int iy;                     // 定义
    std::cout << iy << std::endl;
    //extern int iz;            // 函数内会报错
    std::cout << iz << std::endl;

    return 0;
}

    练习 2.14

    下面的程序合法吗？如果合法，它将输出什么？

    #include <iostream>

// for 循环内i是for循环的局部变量

int main()
{
    int i = 100, sum = 0;
    for (int i = 0; i != 10; ++i)
    {
        sum += i;
    }

    std::cout << i << " " << sum << std::endl;
    return 0;
}
